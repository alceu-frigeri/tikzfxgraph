\documentclass[10pt]{article}

\begin{document}
\ExplSyntaxOn
\keys_define:nn {my module / mod A}
  {  
    A key .usage:n = general ,
    A key .code:n = {A~ key} ,
    A key .groups:n = {group A} ,
    
    AB key .usage:n = general ,
    AB key .code:n = {AB~ key} ,
    AB key .groups:n = {group A} ,

    B key .usage:n = general ,
    B key .code:n = {B~ key} ,
    B key .groups:n = {group B} ,
    
    BB key .usage:n = general ,
    BB key .code:n = {BB~ key} ,
    BB key .groups:n = {group B} ,
  }

  
  
A~ of~ A~ (general~ case):  
\keys_set:nn { my module / mod A}  {A key}\par

B~ of~ A~ (general~ case):  
\keys_set:nn { my module / mod A}  {B key}\par

A~ of~ A~ (group A):  
\keys_set_groups:nnn { my module / mod A} {group A}  {A key}\par

B~ of~ A~ (group A):  
\keys_set_groups:nnn { my module / mod A} {group A}  {B key}\par


A~ of~ A~ (excluding~ group B):  
\keys_set_exclude_groups:nnn { my module / mod A} {group B}  {A key}\par

B~ of~ A~ (excluding~ group B):  
\keys_set_exclude_groups:nnn { my module / mod A} {group B}  {B key}\par


\ExplSyntaxOff
\end{document} 

\NewDocumentCommand{\usrcmd}{m}
  {
    \keys_set_exclude_groups:nnn { my module / mod A} {group B}  {#1}
    ...
  }
